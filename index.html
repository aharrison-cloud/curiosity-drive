/ ============================================================ // NLCS JEJU CURIOSITY DRIVE — Google Apps Script Backend // Paste this entire file into your Apps Script project // ============================================================ const SHEET_NAME_STUDENTS = 'Students'; const SHEET_NAME_ACTIVITY = 'ActivityLog'; const SHEET_NAME_SESSIONS = 'LiveSessions'; // ── CORS helper ────────────────────────────────────────────── function setCORSHeaders(output) { return output .setMimeType(ContentService.MimeType.JSON); } function respond(data) { return setCORSHeaders( ContentService.createTextOutput(JSON.stringify(data)) ); } // ── Router ─────────────────────────────────────────────────── function doGet(e) { const action = e.parameter.action; try { if (action === 'getStudent') return respond(getStudent(e.parameter.email)); if (action === 'getAllStudents') return respond(getAllStudents()); if (action === 'getActivityLog') return respond(getActivityLog(e.parameter.email)); if (action === 'getLiveSession') return respond(getLiveSession()); return respond({ error: 'Unknown action: ' + action }); } catch(err) { return respond({ error: err.message }); } } function doPost(e) { const body = JSON.parse(e.postData.contents); const action = body.action; try { if (action === 'upsertStudent') return respond(upsertStudent(body.data)); if (action === 'logActivity') return respond(logActivity(body.data)); if (action === 'updateLiveSession')return respond(updateLiveSession(body.data)); return respond({ error: 'Unknown action: ' + action }); } catch(err) { return respond({ error: err.message }); } } // ── Sheet helpers ───────────────────────────────────────────── function getOrCreateSheet(name, headers) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.appendRow(headers); sheet.getRange(1, 1, 1, headers.length) .setFontWeight('bold') .setBackground('#002046') .setFontColor('#C5A059'); } return sheet; } function sheetToObjects(sheet) { const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; const headers = data[0]; return data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); } // ── Students ────────────────────────────────────────────────── const STUDENT_HEADERS = [ 'email','name','picture','role', 'xp','level','badges', 'mastery','avatarConfig', 'homeworkPending','helpUsageCount', 'currentGrade','predictedGrade', 'lastSeen','createdAt' ]; function getStudent(email) { if (!email) return { error: 'No email provided' }; const sheet = getOrCreateSheet(SHEET_NAME_STUDENTS, STUDENT_HEADERS); const rows = sheetToObjects(sheet); const found = rows.find(r => r.email === email); if (!found) return { student: null }; // Parse JSON fields found.mastery = safeJSON(found.mastery, {}); found.avatarConfig = safeJSON(found.avatarConfig, null); found.badges = safeJSON(found.badges, []); return { student: found }; } function getAllStudents() { const sheet = getOrCreateSheet(SHEET_NAME_STUDENTS, STUDENT_HEADERS); const rows = sheetToObjects(sheet); const students = rows.map(s => ({ ...s, mastery: safeJSON(s.mastery, {}), avatarConfig: safeJSON(s.avatarConfig, null), badges: safeJSON(s.badges, []) })); return { students }; } function upsertStudent(data) { const sheet = getOrCreateSheet(SHEET_NAME_STUDENTS, STUDENT_HEADERS); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const emailCol = headers.indexOf('email'); // Serialise JSON fields const row = STUDENT_HEADERS.map(h => { const val = data[h]; if (val === undefined || val === null) return ''; if (typeof val === 'object') return JSON.stringify(val); return val; }); // Find existing row for (let i = 1; i < rows.length; i++) { if (rows[i][emailCol] === data.email) { sheet.getRange(i + 1, 1, 1, row.length).setValues([row]); return { success: true, action: 'updated' }; } } // New student row[STUDENT_HEADERS.indexOf('createdAt')] = new Date().toISOString(); sheet.appendRow(row); return { success: true, action: 'created' }; } // ── Activity Log ────────────────────────────────────────────── const ACTIVITY_HEADERS = [ 'id','email','studentName','topicId','topicName', 'mode','score','usedHelp','grade','timestamp' ]; function logActivity(data) { const sheet = getOrCreateSheet(SHEET_NAME_ACTIVITY, ACTIVITY_HEADERS); const row = ACTIVITY_HEADERS.map(h => { if (h === 'timestamp') return new Date().toISOString(); if (h === 'id') return Utilities.getUuid(); return data[h] !== undefined ? data[h] : ''; }); sheet.appendRow(row); return { success: true }; } function getActivityLog(email) { const sheet = getOrCreateSheet(SHEET_NAME_ACTIVITY, ACTIVITY_HEADERS); const rows = sheetToObjects(sheet); const log = email ? rows.filter(r => r.email === email) : rows; return { log: log.slice(-200) }; // last 200 entries } // ── Live Session ────────────────────────────────────────────── const SESSION_HEADERS = [ 'id','isActive','content','questions', 'currentStage','studentResponses','updatedAt' ]; function getLiveSession() { const sheet = getOrCreateSheet(SHEET_NAME_SESSIONS, SESSION_HEADERS); const rows = sheetToObjects(sheet); if (rows.length === 0) { return { session: { isActive: false, content: '', questions: [], currentStage: 'WAITING', studentResponses: [] } }; } const s = rows[rows.length - 1]; return { session: { ...s, isActive: s.isActive === 'true' || s.isActive === true, questions: safeJSON(s.questions, []), studentResponses: safeJSON(s.studentResponses, []) }}; } function updateLiveSession(data) { const sheet = getOrCreateSheet(SHEET_NAME_SESSIONS, SESSION_HEADERS); const rows = sheet.getDataRange().getValues(); const row = SESSION_HEADERS.map(h => { if (h === 'updatedAt') return new Date().toISOString(); if (h === 'id') return rows.length > 1 ? rows[rows.length-1][0] : Utilities.getUuid(); const val = data[h]; if (val === undefined || val === null) return ''; if (typeof val === 'object') return JSON.stringify(val); return val; }); if (rows.length > 1) { // Update last row sheet.getRange(rows.length, 1, 1, row.length).setValues([row]); } else { sheet.appendRow(row); } return { success: true }; } // ── Utility ─────────────────────────────────────────────────── function safeJSON(val, fallback) { if (!val || val === '') return fallback; if (typeof val === 'object') return val; try { return JSON.parse(val); } catch(e) { return fallback; } }
